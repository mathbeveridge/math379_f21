For any \(n \in \mathbb{N}\text{,}\)

\begin{equation*} \begin{array}{rcccl} \displaystyle{ (1+x)^n } &=& \displaystyle{ \sum_{k=0}^{n} {n \choose k} x^k }&=& \displaystyle{ 1 + {n \choose 1} x + {n \choose 2} x^2 + {n \choose 3} x^3 + \cdots + {n \choose n} x^n } \\ \displaystyle{ (1+ax)^n } &=& \displaystyle{ \sum_{k=0}^{n} {n \choose k} a^k x^k }&=& \displaystyle{ 1 + {n \choose 1} a x + {n \choose 2} a^2 x^2 + {n \choose 3} a^3 x^3 + \cdots + {n \choose n} a^n x^n } \\ \displaystyle{ (1+x^r)^n } &=& \displaystyle{ \sum_{k=0}^{n} {n \choose k} x^{kr} }&=& \displaystyle{ 1 + {n \choose 1} x^r + {n \choose 2} x^{2r} + {n \choose 3} x^{3r} + \cdots + {n \choose n} x^{nr} } \\ \end{array} \end{equation*}
