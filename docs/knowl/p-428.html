For any \(n \in \mathbb{N}\text{,}\)

\begin{equation*} \begin{array}{rcccl} \displaystyle{ (1+x)^k } &=& \displaystyle{ \sum_{n=0}^{k} {k \choose n} x^n }&=& \displaystyle{ 1 + {k \choose 1} x + {k \choose 2} x^2 + {k \choose 3} x^3 + \cdots + {k \choose k} x^k } \\ \displaystyle{ (1+ax)^k } &=& \displaystyle{ \sum_{n=0}^{k} {k \choose n} a^n x^n }&=& \displaystyle{ 1 + {k \choose 1} a x + {k \choose 2} a^2 x^2 + {k \choose 3} a^3 x^3 + \cdots + {k \choose k} a^k x^k } \\ \displaystyle{ (1+x^r)^k } &=& \displaystyle{ \sum_{n=0}^{k} {k \choose n} x^{nr} }&=& \displaystyle{ 1 + {k \choose 1} x^r + {k \choose 2} x^{2r} + {k \choose 3} x^{3r} + \cdots + {k \choose k} x^{kr} } \\ \end{array} \end{equation*}
